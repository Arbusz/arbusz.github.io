一个遍历目录的小程序

5 July, 2018

Tags:
  • golang
  • 代码

package main

import (
	"fmt"
	"io/ioutil"
)

func main() {
	folder := "."
	listFile(folder)
}

func listFile(folder string) {
	files, _ := ioutil.ReadDir(folder) //specify the current dir
	for _, file := range files {
		if file.IsDir() {
			listFile(folder + "/" + file.Name())
		} else {
			fmt.Println(folder + "/" + file.Name())
		}
	}

}

递归打印当前目录下所有文件名。
